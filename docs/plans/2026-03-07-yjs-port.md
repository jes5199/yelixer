# Yjs CRDT Port to Idiomatic Elixir — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Port the Yjs/YATA CRDT to idiomatic Elixir, producing a library that can create, edit, encode, decode, and merge collaborative documents with full compatibility with the Yjs binary protocol.

**Architecture:** The port follows the Rust (yrs) implementation more closely than the JS one, since yrs has cleaner separation of concerns and its data-oriented design maps better to Elixir's immutable data structures. The core insight is that Yjs's mutable linked-list Items become an immutable block store indexed by client ID, and transactions become functional transformations over document state. We use Elixir structs, pattern matching, and protocols rather than OOP inheritance.

**Tech Stack:** Elixir 1.18.4, OTP 27, ExUnit, no external dependencies initially. Binary encoding uses Elixir bitstring matching. Property-based testing with StreamData for CRDT invariants.

**Reference repos:**
- `/home/jes/yelixer/yjs/` — JavaScript implementation (v14 RC)
- `/home/jes/yelixer/y-crdt/` — Rust implementation (yrs v0.25)

## Design Decisions

### Why follow yrs over yjs?

1. **Data-oriented:** yrs uses a unified `Branch` struct projected into types via `TypeRef` — maps cleanly to Elixir structs
2. **Explicit transactions:** yrs has `Transaction` / `TransactionMut` — maps to functional state transformations
3. **No mutation:** yrs's `BlockStore` with per-client sorted vectors maps to Elixir maps/lists naturally
4. **Type safety:** yrs's enum-based `ItemContent` maps to tagged tuples or sum types via pattern matching

### Elixir-idiomatic adaptations

- **No linked list pointers:** Replace `left`/`right` Item pointers with positional lookups in sorted block lists. Items store `origin`/`right_origin` IDs (the algorithm only needs these for integration).
- **Immutable state:** `Doc` is a struct passed through functions, not a mutable object. Transactions are `Doc -> Doc` transformations.
- **Protocols:** Use Elixir protocols for `Encodable`/`Decodable` rather than OOP inheritance.
- **Pattern matching:** `ItemContent` variants become tagged tuples: `{:string, "hello"}`, `{:deleted, 5}`, etc.
- **Processes optional:** The core library is pure functional. A `GenServer` wrapper for a "live doc" is a separate, optional layer.

### What to port vs skip (Phase 1)

**Port:**
- ID, StateVector, DeleteSet (core primitives)
- Item, GC, Block storage
- YATA integration algorithm
- Doc, Transaction
- YArray, YMap, YText (core types)
- V1 update encoding/decoding (for interop testing)

**Skip for now:**
- XML types, UndoManager, RelativePosition, Attribution
- V2 encoding, Snapshots
- Awareness, Subdocs, WeakRefs
- Move operations

```elixir
# lib/yelixer/state_vector.ex
defmodule Yelixer.StateVector do
  @type t :: %__MODULE__{clocks: %{non_neg_integer() => non_neg_integer()}}
  defstruct clocks: %{}

  def new, do: %__MODULE__{}

  def get(%__MODULE__{clocks: clocks}, client) do
    Map.get(clocks, client, 0)
  end

  def set(%__MODULE__{clocks: clocks}, client, clock) do
    %__MODULE__{clocks: Map.put(clocks, client, clock)}
  end

  def advance(%__MODULE__{} = sv, client, clock) do
    if clock > get(sv, client), do: set(sv, client, clock), else: sv
  end

  def diff(%__MODULE__{clocks: remote}, %__MODULE__{clocks: local}) do
    remote
    |> Enum.reduce(%{}, fn {client, remote_clock}, acc ->
      local_clock = Map.get(local, client, 0)
      if remote_clock > local_clock do
        Map.put(acc, client, local_clock)
      else
        acc
      end
    end)
  end
end
```

```elixir
# lib/yelixer/delete_set.ex
defmodule Yelixer.DeleteSet do
  @moduledoc """
  Tracks deleted block ranges per client as sorted, non-overlapping intervals.
  """

  @type range :: {non_neg_integer(), non_neg_integer()}
  @type t :: %__MODULE__{clients: %{non_neg_integer() => [range()]}}
  defstruct clients: %{}

  def new, do: %__MODULE__{}

  def insert(%__MODULE__{clients: clients}, client, clock, len) do
    ranges = Map.get(clients, client, [])
    ranges = add_range(ranges, {clock, clock + len})
    %__MODULE__{clients: Map.put(clients, client, ranges)}
  end

  def deleted?(%__MODULE__{clients: clients}, client, clock) do
    clients
    |> Map.get(client, [])
    |> Enum.any?(fn {start, stop} -> clock >= start and clock < stop end)
  end

  def merge(%__MODULE__{clients: c1}, %__MODULE__{clients: c2}) do
    merged =
      Map.merge(c1, c2, fn _client, ranges1, ranges2 ->
        Enum.reduce(ranges2, ranges1, fn range, acc -> add_range(acc, range) end)
      end)

    %__MODULE__{clients: merged}
  end

  defp add_range(ranges, {new_start, new_end}) do
    {overlapping, rest} =
      Enum.split_with(ranges, fn {s, e} ->
        s <= new_end and new_start <= e
      end)

    merged_start = Enum.reduce(overlapping, new_start, fn {s, _}, acc -> min(s, acc) end)
    merged_end = Enum.reduce(overlapping, new_end, fn {_, e}, acc -> max(e, acc) end)

    [{merged_start, merged_end} | rest]
    |> Enum.sort()
  end
end
```

```elixir
# lib/yelixer/item.ex
defmodule Yelixer.Item do
  alias Yelixer.ID

  @type content ::
          {:string, String.t()}
          | {:any, list()}
          | {:binary, binary()}
          | {:deleted, non_neg_integer()}
          | {:embed, term()}
          | {:format, {String.t(), term()}}
          | {:type, atom()}
          | {:json, list()}
          | {:doc, term()}

  @type parent_ref :: {:named, String.t()} | {:id, ID.t()}

  @type t :: %__MODULE__{
          id: ID.t(),
          origin: ID.t() | nil,
          right_origin: ID.t() | nil,
          content: content(),
          parent: parent_ref(),
          parent_sub: String.t() | nil,
          deleted: boolean(),
          length: non_neg_integer()
        }

  defstruct [:id, :origin, :right_origin, :content, :parent, :parent_sub, :deleted, :length]

  def new(id, origin, right_origin, content, parent, parent_sub) do
    %__MODULE__{
      id: id,
      origin: origin,
      right_origin: right_origin,
      content: content,
      parent: parent,
      parent_sub: parent_sub,
      deleted: match?({:deleted, _}, content),
      length: content_length(content)
    }
  end

  defp content_length({:string, s}), do: String.length(s)
  defp content_length({:any, list}), do: length(list)
  defp content_length({:binary, b}), do: byte_size(b)
  defp content_length({:deleted, n}), do: n
  defp content_length({:json, list}), do: length(list)
  defp content_length(_), do: 1
end
```

```elixir
# lib/yelixer/block_store.ex
defmodule Yelixer.BlockStore do
  @moduledoc """
  Stores integrated blocks (Items and GC) organized by client ID.
  Each client's blocks are kept in a list sorted by clock.
  """

  alias Yelixer.{ID, Item, StateVector}

  @type t :: %__MODULE__{clients: %{non_neg_integer() => [Item.t()]}}
  defstruct clients: %{}

  def new, do: %__MODULE__{}

  def push(%__MODULE__{clients: clients}, %Item{} = item) do
    client_blocks = Map.get(clients, item.id.client, [])
    %__MODULE__{clients: Map.put(clients, item.id.client, client_blocks ++ [item])}
  end

  def get(%__MODULE__{clients: clients}, %ID{client: client, clock: clock}) do
    case Map.get(clients, client) do
      nil -> nil
      blocks -> find_block(blocks, clock)
    end
  end

  def state_vector(%__MODULE__{clients: clients}) do
    Enum.reduce(clients, StateVector.new(), fn {client, blocks}, sv ->
      case List.last(blocks) do
        nil -> sv
        %Item{id: id, length: len} -> StateVector.set(sv, client, id.clock + len)
      end
    end)
  end

  defp find_block(blocks, clock) do
    Enum.find(blocks, fn %Item{id: id, length: len} ->
      clock >= id.clock and clock < id.clock + len
    end)
  end
end
```

```elixir
# lib/yelixer/doc.ex
defmodule Yelixer.Doc do
  alias Yelixer.{BlockStore, DeleteSet}

  @type t :: %__MODULE__{
          client_id: non_neg_integer(),
          store: BlockStore.t(),
          delete_set: DeleteSet.t(),
          types: %{String.t() => atom()}
        }

  defstruct [:client_id, :store, :delete_set, :types]

  def new(opts \\ []) do
    client_id = Keyword.get(opts, :client_id, :rand.uniform(1_000_000_000))

    %__MODULE__{
      client_id: client_id,
      store: BlockStore.new(),
      delete_set: DeleteSet.new(),
      types: %{}
    }
  end

  def has_type?(%__MODULE__{types: types}, name), do: Map.has_key?(types, name)

  def get_or_create_type(%__MODULE__{types: types} = doc, name, type_ref) do
    if Map.has_key?(types, name) do
      {doc, Map.get(types, name)}
    else
      doc = %{doc | types: Map.put(types, name, type_ref)}
      {doc, type_ref}
    end
  end
end
```

```elixir
# lib/yelixer/transaction.ex
defmodule Yelixer.Transaction do
  alias Yelixer.{Doc, ID, Item, BlockStore, DeleteSet, StateVector}

  @type t :: %__MODULE__{
          doc: Doc.t(),
          before_state: StateVector.t(),
          delete_set: DeleteSet.t()
        }

  defstruct [:doc, :before_state, :delete_set]

  def transact(%Doc{} = doc, fun) when is_function(fun, 1) do
    txn = %__MODULE__{
      doc: doc,
      before_state: BlockStore.state_vector(doc.store),
      delete_set: DeleteSet.new()
    }

    {txn, result} = fun.(txn)
    doc = commit(txn)
    {result, doc}
  end

  def insert(%__MODULE__{doc: doc} = txn, parent, parent_sub, content) do
    clock = StateVector.get(BlockStore.state_vector(doc.store), doc.client_id)
    id = ID.new(doc.client_id, clock)
    item = Item.new(id, nil, nil, content, parent, parent_sub)
    store = BlockStore.push(doc.store, item)
    %{txn | doc: %{doc | store: store}}
  end

  defp commit(%__MODULE__{doc: doc, delete_set: ds}) do
    %{doc | delete_set: DeleteSet.merge(doc.delete_set, ds)}
  end
end
```

**Step 3: Implement the YATA integration algorithm**

This is the most complex piece. The implementation should follow the algorithm from yrs `block.rs`:

1. Find the position where the item should go based on its `origin` (left neighbor at creation)
2. If there are conflicts (multiple items with the same origin), scan right and use `right_origin` + client ID comparison to find the exact position
3. Insert the item into the block store at the resolved position

```elixir
# lib/yelixer/integrate.ex
defmodule Yelixer.Integrate do
  @moduledoc """
  YATA integration algorithm — resolves where new items go in the sequence.
  """

  alias Yelixer.{ID, Item, BlockStore}

  def integrate(%BlockStore{} = store, %Item{} = item) do
    store = BlockStore.push(store, item)
    {:ok, store}
  end

  def sequence(%BlockStore{} = store, type_name) do
    store.clients
    |> Enum.flat_map(fn {_client, items} ->
      Enum.filter(items, fn %Item{parent: parent, deleted: deleted} ->
        parent == {:named, type_name} and not deleted
      end)
    end)
    |> sort_by_yata(store)
    |> Enum.flat_map(&content_values/1)
  end

  defp sort_by_yata(items, _store) do
    # Group items by their origin to detect conflicts
    # Items with no conflicts keep their natural order
    # Items with same origin are sorted by client_id (lower first)
    Enum.sort(items, fn a, b ->
      cond do
        # If a's origin is b's id, a comes after b
        a.origin != nil and a.origin == b.id -> false
        # If b's origin is a's id, b comes after a
        b.origin != nil and b.origin == a.id -> true
        # Same origin — compare client ids
        a.origin == b.origin -> a.id.client < b.id.client
        # Different origins — compare by clock within same client, else by origin position
        a.id.client == b.id.client -> a.id.clock < b.id.clock
        true -> a.id.client < b.id.client
      end
    end)
  end

  defp content_values(%Item{content: {:string, s}}), do: [s]
  defp content_values(%Item{content: {:any, list}}), do: list
  defp content_values(%Item{content: {:deleted, _}}), do: []
  defp content_values(%Item{content: content}), do: [content]
end
```

```elixir
def integrate(%BlockStore{} = store, %Item{} = item) do
  type_name = case item.parent do
    {:named, name} -> name
    _ -> nil
  end

  if type_name do
    index = find_insertion_index(store, item, type_name)
    store = BlockStore.insert_at(store, type_name, index, item)
    {:ok, store}
  else
    store = BlockStore.push(store, item)
    {:ok, store}
  end
end

defp find_insertion_index(store, item, type_name) do
  seq_ids = Map.get(store.sequences, type_name, [])

  # Find position after origin
  start_index =
    case item.origin do
      nil -> 0
      origin_id ->
        case Enum.find_index(seq_ids, &(&1 == origin_id)) do
          nil -> 0
          idx -> idx + 1
        end
    end

  # Find position before right_origin
  end_index =
    case item.right_origin do
      nil -> length(seq_ids)
      ro_id ->
        case Enum.find_index(seq_ids, &(&1 == ro_id)) do
          nil -> length(seq_ids)
          idx -> idx
        end
    end

  # If no conflicts, insert at start_index
  if start_index >= end_index do
    start_index
  else
    # Scan through conflicting items and resolve by client ID
    resolve_conflicts(store, item, seq_ids, start_index, end_index)
  end
end

defp resolve_conflicts(_store, item, _seq_ids, start_index, end_index)
     when start_index >= end_index do
  start_index
end

defp resolve_conflicts(store, item, seq_ids, index, end_index) do
  if index >= end_index do
    index
  else
    other_id = Enum.at(seq_ids, index)
    other = BlockStore.get(store, other_id)

    cond do
      other == nil ->
        index

      other.origin == item.origin ->
        # Same origin — lower client ID goes first
        if item.id.client < other.id.client do
          index
        else
          resolve_conflicts(store, item, seq_ids, index + 1, end_index)
        end

      true ->
        index
    end
  end
end
```

```elixir
# lib/yelixer/encoding.ex
defmodule Yelixer.Encoding do
  alias Yelixer.StateVector

  # Unsigned varint encoding (LEB128)
  def encode_uint(n) when n < 128, do: <<n>>

  def encode_uint(n) do
    <<1::1, n::7, encode_uint(Bitwise.bsr(n, 7))::binary>>
  end

  def decode_uint(binary), do: decode_uint(binary, 0, 0)

  defp decode_uint(<<0::1, value::7, rest::binary>>, acc, shift) do
    {acc + Bitwise.bsl(value, shift), rest}
  end

  defp decode_uint(<<1::1, value::7, rest::binary>>, acc, shift) do
    decode_uint(rest, acc + Bitwise.bsl(value, shift), shift + 7)
  end

  # String encoding: varint length + UTF-8 bytes
  def encode_string(s) do
    bytes = :erlang.iolist_to_binary(s)
    <<encode_uint(byte_size(bytes))::binary, bytes::binary>>
  end

  def decode_string(binary) do
    {len, rest} = decode_uint(binary)
    <<s::binary-size(len), rest2::binary>> = rest
    {s, rest2}
  end

  # State vector: varint count, then (client, clock) pairs
  def encode_state_vector(%StateVector{clocks: clocks}) do
    count = map_size(clocks)
    pairs = Enum.reduce(clocks, <<>>, fn {client, clock}, acc ->
      <<acc::binary, encode_uint(client)::binary, encode_uint(clock)::binary>>
    end)
    <<encode_uint(count)::binary, pairs::binary>>
  end

  def decode_state_vector(binary) do
    {count, rest} = decode_uint(binary)
    decode_sv_pairs(rest, count, StateVector.new())
  end

  defp decode_sv_pairs(rest, 0, sv), do: sv

  defp decode_sv_pairs(binary, remaining, sv) do
    {client, rest} = decode_uint(binary)
    {clock, rest} = decode_uint(rest)
    decode_sv_pairs(rest, remaining - 1, StateVector.set(sv, client, clock))
  end
end
```

## Task 11: Cross-Peer Sync Integration Test

**Files:**
- Test: `test/yelixer/sync_test.exs`

**Context:** The ultimate validation — two docs make independent edits, exchange updates, and converge to the same state. This exercises the full stack: transactions, YATA integration, encoding, decoding.

**Step 1: Write the integration test**

```elixir
# test/yelixer/sync_test.exs
defmodule Yelixer.SyncTest do
  use ExUnit.Case, async: true

  alias Yelixer.{Doc, Types.Text, Encoding}

  test "two peers sync text edits and converge" do
    doc1 = Doc.new(client_id: 1)
    doc2 = Doc.new(client_id: 2)

    {doc1, _} = Doc.get_or_create_type(doc1, "text", :text)
    {doc2, _} = Doc.get_or_create_type(doc2, "text", :text)

    # Peer 1 types "hello "
    doc1 = Text.insert(doc1, "text", 0, "hello ")

    # Peer 2 types "world"
    doc2 = Text.insert(doc2, "text", 0, "world")

    # Exchange updates
    update1 = Encoding.encode_update(doc1)
    update2 = Encoding.encode_update(doc2)

    {:ok, doc1} = Encoding.apply_update(doc1, update2)
    {:ok, doc2} = Encoding.apply_update(doc2, update1)

    # Both docs should have the same text (order determined by client ID)
    text1 = Text.to_string(doc1, "text")
    text2 = Text.to_string(doc2, "text")
    assert text1 == text2
    # Client 1 < Client 2, so "hello " comes before "world"
    assert text1 == "hello world"
  end

  test "three-way sync converges" do
    doc1 = Doc.new(client_id: 1)
    doc2 = Doc.new(client_id: 2)
    doc3 = Doc.new(client_id: 3)

    for doc <- [doc1, doc2, doc3] do
      {doc, _} = Doc.get_or_create_type(doc, "text", :text)
      doc
    end
    |> then(fn [d1, d2, d3] ->
      d1 = Text.insert(d1, "text", 0, "A")
      d2 = Text.insert(d2, "text", 0, "B")
      d3 = Text.insert(d3, "text", 0, "C")

      u1 = Encoding.encode_update(d1)
      u2 = Encoding.encode_update(d2)
      u3 = Encoding.encode_update(d3)

      {:ok, d1} = Encoding.apply_update(d1, u2)
      {:ok, d1} = Encoding.apply_update(d1, u3)
      {:ok, d2} = Encoding.apply_update(d2, u1)
      {:ok, d2} = Encoding.apply_update(d2, u3)
      {:ok, d3} = Encoding.apply_update(d3, u1)
      {:ok, d3} = Encoding.apply_update(d3, u2)

      assert Text.to_string(d1, "text") == Text.to_string(d2, "text")
      assert Text.to_string(d2, "text") == Text.to_string(d3, "text")
    end)
  end
end
```

```elixir
# test/yelixer/properties_test.exs
defmodule Yelixer.PropertiesTest do
  use ExUnit.Case, async: true
  use ExUnitProperties

  alias Yelixer.{Doc, Types.Text, Encoding}

  property "two peers always converge regardless of operation order" do
    check all ops1 <- list_of(text_op_gen(), min_length: 1, max_length: 10),
              ops2 <- list_of(text_op_gen(), min_length: 1, max_length: 10) do
      doc1 = setup_doc(1, ops1)
      doc2 = setup_doc(2, ops2)

      u1 = Encoding.encode_update(doc1)
      u2 = Encoding.encode_update(doc2)

      {:ok, doc1} = Encoding.apply_update(doc1, u2)
      {:ok, doc2} = Encoding.apply_update(doc2, u1)

      assert Text.to_string(doc1, "text") == Text.to_string(doc2, "text")
    end
  end

  property "applying same update twice is idempotent" do
    check all ops <- list_of(text_op_gen(), min_length: 1, max_length: 5) do
      doc1 = setup_doc(1, ops)
      update = Encoding.encode_update(doc1)

      doc2 = Doc.new(client_id: 2)
      {doc2, _} = Doc.get_or_create_type(doc2, "text", :text)

      {:ok, doc2} = Encoding.apply_update(doc2, update)
      text_once = Text.to_string(doc2, "text")

      {:ok, doc2} = Encoding.apply_update(doc2, update)
      text_twice = Text.to_string(doc2, "text")

      assert text_once == text_twice
    end
  end

  defp text_op_gen do
    gen all text <- string(:alphanumeric, min_length: 1, max_length: 5) do
      {:insert, text}
    end
  end

  defp setup_doc(client_id, ops) do
    doc = Doc.new(client_id: client_id)
    {doc, _} = Doc.get_or_create_type(doc, "text", :text)

    Enum.reduce(ops, doc, fn {:insert, text}, doc ->
      len = Text.length(doc, "text")
      pos = if len == 0, do: 0, else: :rand.uniform(len) - 1
      Text.insert(doc, "text", pos, text)
    end)
  end
end
```
